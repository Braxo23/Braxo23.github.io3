import pandas as pd from sklearn.preprocessing import StandardScaler from sklearn.cluster import KMeans import seaborn as sns # Aggregate activities by branch & category pivot = (df .groupby(["component", "category"])["count"] .sum().unstack("category").fillna(0) ) # K-Means clustering on risk indicators features = ["referral_rate", "completion_pct", "population_size", "program_age"] X = StandardScaler().fit_transform(df[features]) df["cluster"] = KMeans( n_clusters=4, random_state=42 ).fit_predict(X) # Stacked bar chart pivot.plot(kind="bar", stacked=True, colormap="tab10") # Correlation heatmap sns.heatmap(df[features].corr(), annot=True, cmap="Blues", fmt=".2f")

library(tidyverse) library(ggplot2) library(broom) # Wrangle cohort data df <- read_csv("byteback_cohorts.csv") |> filter(!is.na(placement_rate)) |> mutate(income_lift = post_salary - pre_salary) # OLS: enrollment → placement rate model <- lm( placement_rate ~ enrollment + track + cohort_year, data = df ) tidy(model, conf.int = TRUE) # ggplot2 scatter + regression band ggplot(df, aes( x = enrollment, y = placement_rate, color = cohort_year, size = income_lift )) + geom_point(alpha = 0.75) + geom_smooth( method = "lm", se = TRUE, color = "#00ff9d" ) + scale_y_continuous(labels = percent_format()) + theme_minimal()

import * as d3 from "d3"; // Scales const x = d3.scaleTime() .domain(d3.extent(raw, d => d.date)) .range([0, width]); const y = d3.scaleLinear() .domain([0, d3.max(raw, d => d.enrolled)]) .range([height, 0]); // Area generator with catmull-rom curves const makeArea = key => d3.area() .x( d => x(d.date)) .y0(height) .y1(d => y(d[key])) .curve(d3.curveCatmullRom); // Render stacked area paths ["enrolled", "retained", "placed"] .forEach((k, i) => { svg.append("path") .datum(raw) .attr("d", makeArea(k)) .attr("fill", PALETTE[i]) .attr("opacity", 0.2); });
